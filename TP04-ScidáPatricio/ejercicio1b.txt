ALGORITMO: suma_posiciones_pares_e_impares
ENTRADAS: numeroIngresado: entero, arreglo[20].
SALIDA: sumaPares <- 0, sumaImpares <- 0: enteros
variables aux: i <- 0, k <- 0: enteros positivos

A1.Escribir("Ingrese 20 numeros") 
para (i <- 0; i < 20; i <- i + 1) Hacer
	Leer(numeroIngresado)
	arreglo[i] = numeroIngresado
FIN PARA
A2.
Mientras (k < 20) Hacer
	sumaPares <- sumaPares + arreglo[k]
	k <- k + 2;
Fin mientras

Para (k <- 1; k < 20; k <- k + 1) Hacer
	sumaImpares <- sumaImpares + arreglo[k]
FIN PARA

A3. 
Escribir ("La suma total de los indices pares del arreglo es de: " + sumaPares)
Escribir ("La suma total de los indices impares del arreglo es de: " + sumaImpares)

A4. PARAR
